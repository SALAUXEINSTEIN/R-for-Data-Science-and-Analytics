# Data Preparation
# Separate the headers
header <- df1[1,]
# Select 300 observations for exploratory data analysis
weather <- df1[1851:2150,]

weather <-rbind(header, weather)


head(weather)
tail(weather)













# Transposing the Data Weather
start <- 3
step <- 10

# Initialize output data frame
output <- data.frame(TSK=double(),
                    PSFC=double(),
                    U10=double(),
                    V10=double(),
                    Q2=double(),
                    RAINC=double(),
                    SNOW=double(),
                    TSLB=double(),
                    SMOIS=double(),
                    datetime=character(),
                    stringsAsFactors=FALSE)

location <- data.frame(lat=character(),
                      long=character(),
                      stringsAsFactors=FALSE)
start_row <- 1
stop_row <- nrow(weather)-1
row_range <- stop_row-start_row

# Loop over columns in weather data frame, skipping the first two
for (1 in seq(from=start, to=ncol(weather), by=step)) {
    # Get date from column header
    date <- colnames(weather)[1]
    
    # subset weather data frame to columns for this date
    subset <- weather[2:nrow(weather),1:(1+step-1)]
    
    # Add date column
    subset$date <- date
    
    # Rename columns
    colnames(subset) <- c("TSK","PSFC","U10","V10","Q2","RAINC","SNOW","TSLB","SMOIS", "datetime")
    
    # Add subset to output data frame
    output <- rbind(output, subset)
    
    location[start_row:stop_row,1] <- weather[2:nrow(weather),1]
    location[start_row:stop_row,2] <- weather[2:nrow(weather),2]
    start_row <- stop_row+1
    stop_row <- start_row + row_range
}

output <- cbind(location, output)




colSums(is.na(output))

#interpolation

library(dplyr)

output <- arrange(output, lat, long)

library(imputers)

# convert to numeric
for (col in c("TSK","PSFC","U10","V10","Q2","RAINC","SNOW","TSLB","SMOIS")) {
    output[[col]] <- as.numeric(output[[col]])
}

# Interpolate missing values using periods before and after
# Loop through each unique combination of the lat and long
for (g1 in unique(output$lat)) {
    for (g2 in unique(output$long[output$lat == g1])) {
        # Subset data frame to current combination
        sub_output <- output %>%
          filter(lat == g1 & long ==g2)
        # Interpolate missing values in the column
        for (col in c("TSK","PSFC","U10","V10","Q2","RAINC","SNOW","TSLB","SMOIS")) {
            sub_output[[col]] <- na_interpolation(sub_output[[col]], option = "stine", maxgap = 3)
            # Replace original values with interpolated values in the original data frame
            output[output$lat == g1 & output&long == g2, col] <- sub_output[[col]]
        }
    }
}
colsums(is.na(output))
